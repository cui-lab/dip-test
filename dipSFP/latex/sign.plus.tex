\HeaderA{sign.plus}{Positive trucation function}{sign.plus}
\begin{Description}\relax
Given a vector, its negative elements are replaced by zero.
\end{Description}
\begin{Usage}
\begin{verbatim}
sign.plus(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] A vector.

\end{ldescription}
\end{Arguments}
\begin{Value}
A vector with the same length as \code{x}.
\end{Value}
